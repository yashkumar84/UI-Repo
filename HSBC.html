import java.util.HashMap; public int[] twoSum(int[] nums, int target) { HashMap<Integer, Integer> map = new HashMap<>(); for(int i = 0; i < nums.length; i++) { int complement = target - nums[i]; // Check if complement exists if(map.containsKey(complement)) { return new int[]{map.get(complement), i}; } // Store current number with index map.put(nums[i], i); } return new int[]{}; }

Q1: Kadane's Algorithm - Maximum Subarray Sum

Problem: Continuous subarray mein maximum sum find karo

Example: [-2,1,-3,4,-1,2,1,-5,4] ‚Üí 6 (subarray: [4,-1,2,1])

üîó LeetCode Link

Approach: Dynamic Programming

Har position pe decide: current ko add karu ya nayi subarray shuru karu?

Formula: maxEndingHere = max(nums[i], maxEndingHere + nums[i])

Java
‚è± O(n) üíæ O(1)
public int maxSubArray(int[] nums) { int maxSoFar = nums[0]; int maxEndingHere = nums[0]; for(int i = 1; i < nums.length; i++) { maxEndingHere = Math.max(nums[i], maxEndingHere + nums[i]); maxSoFar = Math.max(maxSoFar, maxEndingHere); } return maxSoFar; }

import java.util.Stack; public boolean isValid(String s) { Stack<Character> stack = new Stack<>(); for(char c : s.toCharArray()) { if(c == '(' || c == '{' || c == '[') { stack.push(c); } else { if(stack.isEmpty()) return false; char top = stack.pop(); if((c == ')' && top != '(') || (c == '}' && top != '{') || (c == ']' && top != '[')) { return false; } } } return stack.isEmpty(); }
Q4: Binary Search (Recursive)

Concept: Sorted array mein O(log n) time mein search

Trick: Middle check ‚Üí left ya right half recurse

üîó LeetCode Link

Overflow Safe Formula

‚ùå mid = (left + right) / 2 ‚Üí Can overflow!

‚úÖ mid = left + (right - left) / 2 ‚Üí Safe approach

Java
‚è± O(log n) üíæ O(log n)
public int binarySearch(int[] arr, int target, int left, int right) { if(left > right) return -1; // Overflow safe mid calculation int mid = left + (right - left) / 2; if(arr[mid] == target) return mid; if(arr[mid] > target) { return binarySearch(arr, target, left, mid - 1); } return binarySearch(arr, target, mid + 1, right); }
